#include <iostream>
#include <string>
using namespace std;

bool isNumericConstant(const string& str) {
    if (str.empty()) return false;

    for (char ch : str) {
        if (!isdigit(ch)) {
            return false;
        }
    }
    return true;
}

int main() {
    string input;
    cout << "Enter : ";
    cin >> input;

    if (isNumericConstant(input)) {
        cout << "numeric constant" << endl;
    } else {
        cout << "not numeric" << endl;
    }

    return 0;
}

#include <iostream>
#include <string>
#include <vector>
using namespace std;

bool isOperator(char ch) {
    return ch == '+' || ch == '-' || ch == '*' || ch == '/' || ch == '%' || ch == '=';
}

int main() {
    string input;
    cout << "Enter input: ";
    getline(cin, input);

    vector<char> operators;
    for (char ch : input) {
        if (isOperator(ch)) {
            operators.push_back(ch);
        }
    }

    if (operators.empty()) {
        cout << "No operators found in the input." << endl;
    } else {
        for (size_t i = 0; i < operators.size(); i++) {
            cout << "operator" << i + 1 << ":  " << operators[i] << endl;
        }
    }

    return 0;
}

#include <iostream>
#include <string>
using namespace std;

int main() {
    string line;

    cout << "Enter lines of code (type 'END' to stop):" << endl;

    while (true) {
        getline(cin, line);

        if (line == "END") {
            break;
        }


        if (line.find("//") == 0) {
            cout << "single line comment." << endl;
        }

        else if (line.find("/*") == 0 && line.find("*/") != string::npos) {
            cout << "Multiple line comment." << endl;
        }
        else {
            cout << "This is NOT a comment line." << endl;
        }
    }

    return 0;
}

#include <iostream>
#include <cctype>

using namespace std;

bool isIdentifier(const string& str) {
    if (str.empty())
        return false;

    if (!(isalpha(str[0]) || str[0] == '_'))
        return false;


    for (size_t i = 1; i < str.length(); i++) {
        if (!(isalnum(str[i]) || str[i] == '_'))
            return false;
    }

    return true;
}

int main() {
    string input;
    cout << "Enter the string to check: ";
    cin >> input;

    if (isIdentifier(input))
        cout << "'" << input << "' is a valid identifier." << endl;
    else
        cout << "'" << input << "' is NOT a valid identifier." << endl;

    return 0;
}
